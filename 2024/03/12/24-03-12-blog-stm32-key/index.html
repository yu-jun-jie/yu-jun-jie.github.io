#include "stdio.h"
#include "string.h"
#include "stdarg.h"
void uart1printf(const char *format, ...)
{
    char str[256];
    va_list args; /*创建变量*/
    va_start(args, format);
    vsnprintf(str, sizeof(str), format, args); /*变量转换*/
    va_end(args); /*释放变量*/
    HAL_UART_Transmit(&huart1, (uint8_t*)str, strlen(str), 100);
}
